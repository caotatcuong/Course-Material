select distinct BOMON.MABM, BOMON.TENBM
from BOMON, GIAOVIEN, THAMGIADT, DETAI, CHUDE
where BOMON.MAKHOA='CNTT' AND BOMON.MABM = GIAOVIEN.MABM AND GIAOVIEN.MAGV = THAMGIADT.MAGV 
AND THAMGIADT.MADT = DETAI.MADT AND DETAI.MACD = CHUDE.MACD AND CHUDE.TENCD = N'Thanh phố Thông Minh'

select GV.MAGV, GV.HOTEN
from GIAOVIEN GV, GIAOVIEN GVQL, THAMGIADT, DETAI, GIAOVIEN GVCN
where GV.GVQLCM = GVQL.MAGV AND GVQL.HOTEN = N'Nguyễn Văn An'
AND GV.MAGV = THAMGIADT.MAGV AND THAMGIADT.MADT = DETAI.MADT AND DETAI.GVCNDT = GVCN.MAGV AND GVCN.HOTEN = N'Nguyễn Văn Tuấn'

select GV.MAGV, GV.HOTEN
from GIAOVIEN GV, BOMON, THAMGIADT, DETAI, GIAOVIEN GVCN, KHOA
where GV.MABM = BOMON.MABM AND BOMON.MAKHOA = 'CNSH' 
AND GV.MAGV = THAMGIADT.MAGV AND THAMGIADT.MADT = DETAI.MADT AND DETAI.GVCNDT = GVCN.MAGV AND GVCN.MAGV = KHOA.TRUONGKHOA AND KHOA.MAKHOA = 'CNTT'

select GV.HOTEN
from GIAOVIEN GV, BOMON, THAMGIADT, DETAI, GIAOVIEN GVCN
where GV.MAGV = BOMON.TRUONGBM 
AND GV.MAGV = THAMGIADT.MAGV AND THAMGIADT.MADT = DETAI.MADT AND DETAI.GVCNDT = GVCN.MAGV AND GVCN.HOTEN = N'Nguyễn Văn An'
AND GV.MAGV = DETAI.GVCNDT

select distinct GV.MAGV, GV.HOTEN
from GIAOVIEN GV, THAMGIADT, DETAI, GIAOVIEN GVCN, BOMON
where GV.MAGV = THAMGIADT.MAGV AND THAMGIADT.MADT = DETAI.MADT 
AND DETAI.GVCNDT = GVCN.MAGV AND GV.MABM = BOMON.MABM AND GVCN.MAGV = BOMON.TRUONGBM
